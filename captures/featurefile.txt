int FindMax(int T[] , int n) {
int i , max=T[0];
for (i=1;i<n ; i++){
	if (T[i]>max)
	max=T[i];
}
return max ;

}



int FindMin(int T[] , int n) {
int i , min=T[0] ;
for (i=0;i<n ; i++){
        if (T[i]<min)
        max=T[i];
}
return min ;

}
